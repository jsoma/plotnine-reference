Left-aligning the title¶

In [20]:
Copied! 
(
    ggplot(df)
    + aes(x='gdp_per_capita', y='life_expectancy')
    + geom_point()
    + labs(title="Life Expectancy vs GDP (per capita)")
    + theme(plot_title=element_text(ha='left', x=0.125))
)
( ggplot(df) + aes(x='gdp_per_capita', y='life_expectancy') + geom_point() + labs(title="Life Expectancy vs GDP (per capita)") + theme(plot_title=element_text(ha='left', x=0.125)) )
